lib/dependencies/resolver.js
'use strict'; var Node = require('./node'); var nodes = {}; module.exports = { hasNode: function(node) { return !!nodes[node]; }, add: function(name, dependencies) { if (!this.hasNode(name)) { nodes[name] = new Node(name, dependencies); } else { nodes[name].update(dependencies); } }, resolve: function(name) { }, get: function() { return nodes; } };
